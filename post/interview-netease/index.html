1. 小球

题目描述:

小明是一个放置类游戏爱好者，他只要盯着某些东西的运作过程，就能获得放松与舒适。所以小明的一个癖好就是盯着他的电脑的屏幕保护程序动画观看。小明的屏幕保护程序动画是这样的，屏幕上会出现一个n行m列的方格矩阵，其中有两类格子，黑色的格子叫做空地，白色的格子叫做墙，矩阵周围的一圈都是墙（即第一行、第一列、第n行、第m列都是墙，其余地方为空地）。然后屏幕保护程序出现的时候，会在随机一个空地上出现一个带颜色的格子叫做小球（会是红绿蓝三种颜色其中一种），随后，这个格子会随机一个方向（左上、右上、右下、左下四个方向其中之一）每秒移动一格。当颜色格子碰到墙时，其碰到墙方向的运动矢量会反向，同时自己格子的颜色会变化（变成红绿蓝其中一种，且必定与原颜色不同）。小明就是能盯着这个小球来回弹跳变色能看一下午。某天小明看得正开心时突然想上厕所，他记住了格子矩阵的状态和小球的当前状态，他想知道从自己离开座位到上完厕所回来后这段时间里，小球的颜色变化了多少次。（关于“碰到”与“变色”的更详细定义与解释，见下方的提示）

输入描述:

每个输入数据包含多个测试点。第一行为测试点的个数Q(Q <= 5)。对于每个测试点，第一行为两个整数N、M(0 < N <= 200，0 < M <= 200)，分别表示格子矩阵有N行M列。同时这个矩阵，是以第一行在上方，第N行在下方，第一列在左边，第M列在右边的样式显示在小明面前的。接下来一行，四个整数X（3<=X<=N-2)，Y(3<=Y<=M-2)，W(0<=W<=3)，T(0<T<1000)。X、Y表示初始时（第0秒）小球处于X行Y列的格子上，W表示小球初始的运动方向，0表示左上方，1表示右上方，2表示右下方，3表示左下方。T表示小明离开座位到上完测试回来需要T秒。

输出描述:

对于每个测试点，输出一行，该行包括一个整数C，表示T秒内（即1秒到T秒之间）小球的颜色变化了C次。(输入输出示例仅供调试，后台判题数据一般不包含示例)

        tot=0;
        while(t--){
            if(dir==0){
                if((y-1==2)&&(x-1!=2)){
                    tot++;

                    tot++;
                    dir=3;
                }

                    tot+=2;
                    dir=2;
                }
                x--,y--;
            }
            else if(dir==1){
                if((y+1==m-1)&&(x-1!=2)){
                    tot++;
                    dir=0;
                }
                else if((y+1!=m-1)&&(x-1==2)){
                    tot++;
                    dir=2;
                }
                else if((y+1==m-1)&&(x-1==2)){
                    tot+=2;
                    dir=3;
                }
                x--,y++;
            }
            else if(dir==2){
                if((y+1==m-1)&&(x+1!=n-1)){
                    tot++;
                    dir=3;
                }
                else if((y+1!=m-1)&&(x+1==n-1)){
                    tot++;
                    dir=1;
                }
                else if((y+1==m-1)&&(x+1==n-1)){
                    tot+=2;
                    dir=0;
                }
                x++,y++;
            }
            else{
            	if((y-1==2)&&(x+1!=n-1)){
                    tot++;
                    dir=2;
                }
                else if((y-1!=2)&&(x+1==n-1)){
                    tot++;
                    dir=0;
                }
                else if((y-1==2)&&(x+1==n-1)){
                    tot+=2;
                    dir=1;
                }
                x++,y--;
			}
		}
		printf("%d\n",tot);
	}
	return 0;
}

#include<bits/stdc++.h>
using namespace std;
int tot;
int T,line;
//注意处理标题,列表和正文中的超链接即可 
int link(string s,int left)
{
	int i=left;
	while(s[i]!=']'){
		tot++;
		i++;
	}
	while(s[i]!=')') i++;
	return i;
}
void title(string s)
{
	int i=0;
	while(s[i]=='#') i++;
	while(s[i]!='\0'){
		if(s[i]=='['){
			int right=link(s,i+1);
			i=right+1;
		}
		else{
			tot++;
			i++;
		}
	}
	return ;
}
void mlist(string s)
{
	int i=1;
	while(s[i]==' ') i++;
	while(s[i]!='\0'){
		if(s[i]=='['){
			int right=link(s,i+1);
			i=right+1;
		}
		else{
			tot++;
			i++;
		} 
	}
	return ;
}

int main()
{
	scanf("%d",&T);
	scanf("%d",&line);
	string s;
	while(T--){		
		tot=0; 
		while(line--){
			cin>>s;
			if(s[0]=='#'){
				title(s);
//				printf("%d\n",tot);		
			}
			else if(s[0]=='+'){
				mlist(s);
				tot++;
//				printf("%d\n",tot);
			}
			else if(s[0]=='['){
				tot+=link(s,1);
			}
			else{
				int i=0;
				while(s[i]!='\0'){
					if(s[i]=='['){
						int right=link(s,i+1);
						i=right+1;
					}
					else{
						tot++;
						i++;
					} 
				}
			}
		}
		printf("%d\n",tot);
	}
}

3. 方块滚动

题目描述:

给定如下的一个正方体，其6个面编号为0到5。其中0的对立面为5，1的对立面为4，2的对立面为3。

该方块可以滚动，但不可以旋转。上图为初始状态，则其向前滚动和向右滚动后的状态如下图所示：

给定一个由格子组成的二维迷宫和起始点，以及该方块在起点的状态，你需要将该方块滚动到终点。方块在滚动时，会将朝下面的编号留在格子上。因此一条从起点到终点的路径，可以由一系列面的编号组成。方块6个面分别有自己的权重值，一条路径的权重，等于构成这条路径的所有面的权重和。下图展示了一条路径，红色数字为方块面的编号，蓝色点为滚动路径。当方块沿着路径滚动到目标点后，会给路径的格子留下面的编号，如下图蓝色数字所示。这条路径的权重为所有蓝色编号对应的方块面的权重和。你的任务是，计算从起点到终点所有路径中，权重最小的路径其权重值为多少。如果不存在一条路径，则输出-1。

#include<bits/stdc++.h>
using namespace std;
int T;
int m,n;
char a[55][55];
int sx,sy,tx,ty;
int we[6];
int dx[4]={-1,1,0,0};
int dy[4]={0,0,-1,1}; 
bool vis[55][55];
int up,vor,recht,tot;
int minn;
bool flag; 
bool can(int x,int y)
{
	if(!vis[x][y]&&a[x][y]!='#'){
		if(x>=0&&x<=m-1&&y>=0&&y<=n-1){
			return true;
		}
	}
	return false;
}
void dfs(int x,int y,int u,int f,int r)
{
	if(x==tx&&y==ty){
		flag=true;
		minn=min(tot+we[5-u],minn);//加上we[5-u] 
		return ;
	}
	for(int i=0;i<=3;i++){
		int nx=x+dx[i];
		int ny=y+dy[i];
		if(can(nx,ny)){
			vis[x][y]=1; 
			tot+=we[5-u];
			if(i==0)      dfs(nx,ny,f,5-u,r);
			else if(i==1) dfs(nx,ny,5-f,u,r);
			else if(i==2) dfs(nx,ny,r,f,5-u);
			else          dfs(nx,ny,5-r,f,u);
			tot-=we[5-u];//及时回溯 
			vis[x][y]=0;
		}
	}
}
int main()
{
	scanf("%d",&T);
	while(T--){
		scanf("%d%d",&m,&n);
		getchar();
		for(int i=0;i<m;i++){
			for(int j=0;j<n;j++)
				scanf("%c",&a[i][j]);
			getchar();//处理换行符 
		}
		scanf("%d%d%d%d",&sx,&sy,&tx,&ty);
		for(int i=0;i<6;i++) scanf("%d",&we[i]);
		up=1,vor=0,recht=2;//上前右三个面信息必须同时使用 
		tot=0;
		flag=false;
		minn=51*51*501; 
		dfs(sx,sy,up,vor,recht);
		if(flag==false) printf("-1\n");
		else printf("%d\n",minn);
	}
	return 0;
}
